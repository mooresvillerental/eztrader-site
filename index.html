import json import time import os import urllib.request import urllib.parse from pathlib import Path TOKEN = "PASTE_YOUR_REAL_TOKEN_HERE" CHANNEL_ID = "-1003818142639" SITE_URL = "https://geteztrader.com" SIGNAL_FILE = "signals/latest_signal.json" TRADE_FILE = "signals/assistant_trade_history.json" STATE_FILE = "ez_autopost_state.json" POLL_SECONDS = 30 def load_json(path): if not os.path.exists(path): return None with open(path, "r", encoding="utf-8") as f: return json.load(f) def save_json(path, data): with open(path, "w", encoding="utf-8") as f: json.dump(data, f, indent=2) def load_state(): if not os.path.exists(STATE_FILE): return { "last_signal_key": None, "last_trade_key": None } try: return load_json(STATE_FILE) except Exception: return { "last_signal_key": None, "last_trade_key": None } def save_state(state): save_json(STATE_FILE, state) def pretty_strategy(raw): s = str(raw or "").upper() if s == "NO_SIGNAL": return "Waiting for Setup" if s == "A_ONLY": return "Strategy A Only" if s == "B_VOL_BREAKOUT": return "Volatility Breakout" if s == "A_TREND_PULLBACK": return "Trend Pullback" if s == "NONE" or s == "": return "Standby" return str(raw) def money(v): try: return "$" + format(float(v), ",.2f") except Exception: return "$0.00" def telegram_post(text): url = f"https://api.telegram.org/bot{TOKEN}/sendMessage" data = urllib.parse.urlencode({ "chat_id": CHANNEL_ID, "text": text }).encode() req = urllib.request.Request(url, data=data) try: with urllib.request.urlopen(req, timeout=20) as resp: _ = resp.read() print("Posted to Telegram") return True except Exception as e: print("Telegram post failed:", e) return False def should_post_signal(signal): action = str(signal.get("action") or "").upper() final_action = str(signal.get("final_action") or "").upper() # Only public-alert actionable signals actionable = {"BUY", "SELL"} if final_action in actionable: return final_action if action in actionable: return action return None def signal_key(signal, action_to_post): ts = signal.get("timestamp") symbol = signal.get("symbol") confidence = signal.get("confidence") strategy = signal.get("strategy") return f"{symbol}|{action_to_post}|{confidence}|{strategy}|{ts}" def trade_key(trade): return f"{trade.get('symbol')}|{trade.get('action')}|{trade.get('price')}|{trade.get('size_usd')}|{trade.get('timestamp')}" def build_signal_message(signal, action_to_post): symbol = signal.get("symbol", "BTC-USD") confidence = signal.get("confidence", "--") strategy = pretty_strategy(signal.get("strategy")) regime = signal.get("regime", "--") trend = signal.get("trend", "--") suggested = float(signal.get("suggested_trade_usd") or 0) parts = [ f"EZTrader {action_to_post} Signal", "", f"Market: {symbol}", f"Action: {action_to_post}", f"Confidence: {confidence}%", f"Strategy: {strategy}", f"Regime: {regime}", f"Trend: {trend}", ] if suggested > 0: parts.append(f"Suggested Size: {money(suggested)}") parts.extend([ "", "Live dashboard:", SITE_URL ]) return "\n".join(parts) def build_trade_message(trade): symbol = trade.get("symbol", "BTC-USD") action = trade.get("action", "TRADE") price = money(trade.get("price", 0)) size = money(trade.get("size_usd", 0)) qty = trade.get("filled_qty", 0) return "\n".join([ "EZTrader Trade Executed", "", f"{action} {symbol}", f"Price: {price}", f"Size: {size}", f"Qty: {qty}", "", "Live dashboard:", SITE_URL ]) def get_latest_trade(trades): if not isinstance(trades, list) or len(trades) == 0: return None return trades[-1] def main(): print("EZTrader auto-marketing started") state = load_state() while True: try: signal = load_json(SIGNAL_FILE) trades = load_json(TRADE_FILE) # Actionable BUY/SELL signal alerts only if signal: action_to_post = should_post_signal(signal) if action_to_post: skey = signal_key(signal, action_to_post) if skey != state.get("last_signal_key"): msg = build_signal_message(signal, action_to_post) if telegram_post(msg): state["last_signal_key"] = skey save_state(state) # Executed trade alerts latest_trade = get_latest_trade(trades) if latest_trade: tkey = trade_key(latest_trade) if tkey != state.get("last_trade_key"): msg = build_trade_message(latest_trade) if telegram_post(msg): state["last_trade_key"] = tkey save_state(state) except Exception as e: print("Loop error:", e) time.sleep(POLL_SECONDS) if __name__ == "__main__": main()
